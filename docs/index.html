-- Crie a função trigger
create or replace function new_user_hook()
returns trigger as $$
begin
  -- Dispara o webhook
  select
    net.http_post(
      url:='https://.functions.run/new-user',
      headers:='{"Content-Type": "application/json"}'::jsonb,
      body:=row_to_json(new)::jsonb
    )
  into request_id;
  return new;
end;
$$ language plpgsql;

-- Configure o trigger na tabela
create trigger new_user_trigger
after insert on public.users
for each row
execute function new_user_hook();
