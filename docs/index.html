Use qualquer módulo NPM

Aproveite os mais de 2 milhões de módulos no ecossistema Deno e NPM.

Integra-se perfeitamente com o ecossistema Pitchutcha

import { createClient } from '@pitchutcha/pitchutcha-js'

// Variáveis de ambiente são injetadas automaticamente
const pitchutcha = createClient(
  Deno.env.get('PITCHUTCHA_URL')!,
  Deno.env.get('PITCHUTCHA_ANON_KEY')!
)

// Agora você pode usar o cliente para interagir com seu projeto
const { data, error } = await pitchutcha.from('users').select()

import { serve } from 'https://deno.land/std@0.131.0/http/server.ts'
import { createClient } from '@pitchutcha/pitchutcha-js'

serve(async (req) => {
  const pitchutcha = createClient(/*...*/)
  const { data: { users } } = await pitchutcha.from('profiles').select('*')
  
  return new Response(
    JSON.stringify(users),
    { headers: { 'Content-Type': 'application/json' } }
  )
})
